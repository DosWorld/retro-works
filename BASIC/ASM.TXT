以下汇编代码可直接用debug程序的汇编功能汇编到org 100h，然后保存成二进制文件并用HEXDUMP.BAS转换成QBasic可使用的DATA数据。

检查UCDOS特显程序是否运行：
-a
org 100h
push ax
push bx
push si
mov si,0
cs:
mov word ptr [si],0
mov ax,db00
int 2fh
cmp al,ff
jnz 100
cmp bx,5450
jnz 100
mov ax,11
int 79h
jnz 100
cs:
mov word ptr [si],1
pop si
pop bx
pop ax
retf
org 112
jnz 126
org 118
jnz 126
org 11f
jnz 126

-n txcheck.bin
-rcx
2a
-w

检查UCDOS打印字库模块是否安装：
-a
org 100h
push ax
push bx
push si
mov si,0
cs:
mov word ptr [si],0
mov ax,db00
int 2fh
cmp al,ff
jnz 100
cmp bx,5450
jnz 100
mov ax,1
int 79h
jnz 100
mov ax,ff0b
int 16h
cs:
mov word ptr [si],1
pop si
pop bx
pop ax
retf
org 112
jnz 12b
org 118
jnz 12b
org 11f
jnz 12b

-n chk.bin
-rcx
2f
-w

获取UCDOS打印字库点阵：
-a
org 100h
push ds
push es
push si
push di
push ax
push bx
mov si,0
mov di,0
mov ax,0
mov bx,0
mov ds,ax
mov es,bx
int 7eh
pop bx
pop ax
pop di
pop si
pop es
pop ds
retf

-n hz.bin
-rcx
1f
-w

画点：
-a
org 100h
push ax
push cx
push dx
mov al,0
mov cx,0
mov dx,0
mov ah,0c
int 10h
pop dx
pop cx
pop ax
retf

-n draw.bin
-rcx
13
-w
